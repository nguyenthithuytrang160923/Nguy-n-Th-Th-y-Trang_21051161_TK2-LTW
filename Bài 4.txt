            <option value="Trẻ em">Trẻ em</option>
            <option value="Người lớn">Người lớn</option>

        <button onclick="calculateBMI()">Tính BMI</button><br>

        <div id="result"></div>

    <script>
        function calculateBMI() {
            var weight = parseFloat(document.getElementById("weight").value);
            var height = parseFloat(document.getElementById("height").value);
            var ageCategory = document.getElementById("ageCategory").value;

            var bmi = weight / (height * height);

            var resultElement = document.getElementById("result");
            resultElement.innerHTML = "Kết quả: " + bmi.toFixed(2);

            if (bmi >= 40) {
                resultElement.innerHTML += "<br>Béo Phì độ 3";

                if (ageCategory === "Già, lớn tuổi") {
                    resultElement.innerHTML += "<br>Hạn chế ăn mỡ động vật";
                }
            } else {
                resultElement.innerHTML += "<br>BMI không thuộc loại Béo Phì";
            }
        }
    </script>
